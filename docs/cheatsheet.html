var nums: [Int] = [1, 2, 3] nums.append(4) // O(1) amortized nums.insert(0, at: 0) // O(n) — 전체 shift let val = nums[2] // O(1) random access nums.removeLast() // O(1) nums.sort() // O(n log n) TimSort

// Ring Buffer 기반 Deque struct Deque<T> { private var buf: [T?]; private var head = 0 mutating func pushFront(_ v: T) { /* head-- mod cap */ } mutating func pushBack(_ v: T) { /* tail++ mod cap */ } mutating func popFront() -> T? { /* head++ */ } mutating func popBack() -> T? { /* tail-- */ } }

06 Binary Tree 노드당 자식 ≤ 2
순회 방식순서복잡도

class TreeNode<T> { var val: T var left: TreeNode?, right: TreeNode? init(_ val: T) { self.val = val } } // In-order traversal func inorder(_ n: TreeNode<Int>?) { guard let n else { return } inorder(n.left); print(n.val); inorder(n.right) }
사용처SwiftUI View 계층, DOM 트리, 수식 파싱, 파일 시스템
주의순회 방식에 따라 결과 다름. 깊은 트리는 재귀 시 스택 오버플로우 위험

func search(_ node: TreeNode<Int>?, _ t: Int) -> TreeNode<Int>? { guard let n = node else { return nil } if t == n.val { return n } return t < n.val ? search(n.left, t) : search(n.right, t) }
사용처범위 검색, 정렬된 데이터 관리, DB 인덱스 기초
주의편향 트리 시 O(n). 실무에선 AVL/Red-Black Tree로 균형 유지 필수

// 배열 기반 — parent: (i-1)/2, children: 2i+1, 2i+2 var heap = [3, 1, 4, 1, 5] func siftUp(_ i: Int) { var c = i while c > 0 && heap[c] < heap[(c-1)/2] { heap.swapAt(c, (c-1)/2); c = (c-1)/2 } }
사용처우선순위 큐, Top-K 문제, Dijkstra, 힙 정렬, 스케줄러

10 Trie 접두어 트리 — 문자열 특화
연산복잡도비고
insertO(m)m = 문자열 길이
searchO(m)정확한 매칭
startsWithO(m)접두어 검색
deleteO(m)리프→루트 정리
class TrieNode { var children: [Character: TrieNode] = [:] var isEnd = false } func insert(_ word: String) { var node = root for ch in word { if node.children[ch] == nil { node.children[ch] = TrieNode() } node = node.children[ch]! } node.isEnd = true }
사용처자동완성, 사전, IP 라우팅, 맞춤법 검사
주의메모리 사용량 큼. 한글은 자모 분리 필요. 압축 Trie(Radix Tree) 고려
